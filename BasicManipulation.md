 * bit 
   * a & 0 = 0
   * a | 0 = a
   * a ^ 0 = a
   * a % all 1s = a 
   * a | all 1s = all 1s
   * a ^ all 1s = ~a
   * a & a = a
   * a | a = a 
   * a ^ a = 0
